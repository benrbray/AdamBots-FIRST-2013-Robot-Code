009    import edu.wpi.first.wpilibj.Joystick;

013    /**
014     *
015     * @author Ben
016     */
017    public class FancyJoystick extends Joystick {
018        //// CONSTANTS -------------------------------------------------------------
019        
020        public static final double DEFAULT_DEAD_ZONE = 0.3;
021        
022        public static final int BUTTON_A        = 1;
023        public static final int BUTTON_B        = 2;
024        public static final int BUTTON_X        = 3;
025        public static final int BUTTON_Y        = 4;
026        public static final int BUTTON_LB       = 5;
027        public static final int BUTTON_RB       = 6;
028        public static final int BUTTON_BACK     = 7;
029        public static final int BUTTON_START    = 8;
030        public static final int BUTTON_LEFTJOY  = 9;
031        public static final int BUTTON_RIGHTJOY = 10;
032        
033        public static final int AXIS_LEFT_X     = 1;    // LEFT -1, RIGHT   1
034        public static final int AXIS_LEFT_Y     = 2;    // UP   -1, DOWN    1
035        public static final int AXIS_TRIGGERS   = 3;
036        public static final int AXIS_RIGHT_X    = 4;    // LEFT -1, RIGHT   1
037        public static final int AXIS_RIGHT_Y    = 5;    // UP   -1, DOWN    1
038        
039        public static final int XBOX_AXES       = 6;
040        public static final int XBOX_BUTTONS    = 10;
041        
042        //// BUTTONS ---------------------------------------------------------------
043        
044        //// AXES ------------------------------------------------------------------
045        
046        private double _deadZone = 0;
047        private double _liveZone = 1;
048        
049        //// CONSTRUCTOR -----------------------------------------------------------
050        
051        public FancyJoystick(int port){
052            super(port, XBOX_BUTTONS, XBOX_AXES);
053            setDeadZone(DEFAULT_DEAD_ZONE);
054        }
055        
056        public FancyJoystick(int port, double deadZone){
057            super(port, XBOX_BUTTONS, XBOX_AXES);
058            setDeadZone(deadZone);
059        }
060        
061        public FancyJoystick(int port, int numChannels, int numAxes){
062            super(port, numChannels, numAxes);
063            setDeadZone(DEFAULT_DEAD_ZONE);
064        }
065        
066        public FancyJoystick(int port, int numChannels, int numAxes, double deadZone){
067            super(port, numChannels, numAxes);
068            setDeadZone(deadZone);
069        }
070        
071        //// INPUT METHODS ---------------------------------------------------------
072        
073        public double getDeadAxis(int axis){
074            return getDeadAxis(axis, _deadZone);
075        }
076        
077        public double getDeadAxis(int axis, double deadZone){
078            double liveZone = 1 - deadZone;
079            double value = this.getRawAxis(axis);
080            
081            if(Math.abs(value) > deadZone){
082                return (value - (deadZone * MathUtils.sign(value))) / liveZone;
083            } else {
084                return 0.0;
085            }
086        }
087        
088        //// GETTER / SETTER -------------------------------------------------------
089        
090        public void setDeadZone(double value){
091            _deadZone = Math.min(1.0, Math.max(0.0, value));
092            _liveZone = 1 - _deadZone;
093        }
094    }
