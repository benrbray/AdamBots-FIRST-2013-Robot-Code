022    /**

024     * functions corresponding to each mode, as described in the IterativeRobot
025     * documentation.
026     * 
027     * RobotMain includes static instances of each behavior class, because they will
028     * be commonly referenced elsewhere, and we don't want to pass them as arguments
029     * all over the place.
030     * 
031     * @author Ben Bray
032     * @author Steven Ploog
033     */
034    public class RobotMain extends IterativeRobot {
035        //// STATIC INSTANCE VARIABLES ---------------------------------------------
036        
037        /** Statically accessible instance of RobotDrive. */
038        public static RobotDrive robotDrive;
039        /** Statically accessible instance of RobotPickup. */
040        public static RobotPickup robotPickup;

043        /** Statically accessible instance of RobotCamera. */
044        public static RobotCamera robotCamera;
045        /** Statically accessible instance of RobotClimb. */
046        public static RobotClimb robotClimb;
047        
048        //// ROBOT LOGIC PHASES ----------------------------------------------------
049        
050        private ILogicPhase _currentLogicPhase = null;
051        private AutonLogic _autonLogic;
052        private TeleopLogic _teleopLogic;
053        private ClimbLogic _climbLogic;
054        
055        //// ITERATIVE ROBOT METHODS -----------------------------------------------
056        
057        /**
058         * This function is run when the robot is first started up and should be
059         * used for any initialization code.
060         */
061        public void robotInit() {
062            
063        }
064        
065        //// AUTONOMOUS ------------------------------------------------------------
066        
067        /**
068         * Initialization code for the autonomous period.
069         */
070        public void autonomousInit() {
071            _autonLogic = new AutonLogic();
072            segueToLogicPhase(_autonLogic);
073        }
074    
075        /**
076         * This function is called periodically during autonomous.
077         */
078        public void autonomousPeriodic() {
079            // Update the Current Logic Phase (should be _autonLogic)
080            _currentLogicPhase.update();
081        }
082        
083        //// TELEOP ----------------------------------------------------------------
084    
085        /**
086         * Initialization code for the teleoperated period.
087         */
088        public void teleopInit() {
089            _teleopLogic = new TeleopLogic();
090            _climbLogic = new ClimbLogic();
091        }
092        
093        /**
094         * This function is called periodically during operator control.
095         */
096        public void teleopPeriodic() {
097            // Update the Current Logic Phase (should be _teleopLogic or _climbLogic)
098            _currentLogicPhase.update();
099        }

133        //// LOGICPHASE METHODS ----------------------------------------------------
134        
135        /**
136         * Revokes power from the logic phase currently in control and grants
137         * control to the phase specified.  Before the segue, this method invokes
138         * finish() in the original phase, and after the segue, this method invokes 
139         * init() in the new phase.
140         * @param phase The phase to transition to.
141         * @return Boolean value indicating the success or failure of the segue.
142         * @see ILogicPhase
143         * @see ILogicPhase#finish() 
144         * @see ILogicPhase#init();
145         */
146        private boolean segueToLogicPhase(ILogicPhase phase){
147            if(_currentLogicPhase != null){
148                _currentLogicPhase.finish();
149            }
150            
151            _currentLogicPhase = phase;
152            _currentLogicPhase.init();
153            
154            return true; // TODO:  Update segueToLogicPhase() return value as needed.
155        }
156    }
