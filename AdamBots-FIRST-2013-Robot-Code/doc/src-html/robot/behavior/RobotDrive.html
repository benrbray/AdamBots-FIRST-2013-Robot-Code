005    package robot.behavior;

007    import edu.wpi.first.wpilibj.*;
008    import robot.actuators.RobotActuators;
009    
010    
011    /**
012     *
013     * @author Ben
014     */
015    public abstract class RobotDrive {
016        
017        // there will be a class TBD that will store all the robot
018        // sensors, motors
019        // take out the next few lines when static class is made
020        public static final int PORT_LEFT_VICTOR = 1;
021        public static final int PORT_RIGHT_VICTOR = 2;
022        /** Shifter value*/
023        public static final double SHIFTER_LOW = 1;
024        public static final double SHIFTER_HIGH = 0.3;
025        public static final double SHIFTER_NEUTRAL = 0.8;
026        
027        //private robot.actuators.RobotActuators robotActuators;
028        
029        /**
030         * Initialize everything
031         */
032        public static void init() {
033        }
034        
035        /**
036         * 
037         * @param leftSpeed to set the left speed
038         * @param rightSpeed to set the right speed
039         * sets the speed of the wheels to the parameters given
040         */
041        public static void drive( double leftSpeed, double rightSpeed ) {
042            RobotActuators.driveRight.set(rightSpeed);
043            RobotActuators.driveLeft.set(leftSpeed);
044        }
045        
046        /**
047         * 
048         * @param speed for the speed of both wheels
049         * calls drive and sends it the parameters of speed for both
050         */
051        public static void driveStraight( double speed )
052        {
053            RobotActuators.driveRight.set(speed);
054            RobotActuators.driveLeft.set(speed);
055        }
056        
057        /**
058         * 
059         * @param speed to turn in place
060         * turns in place at the speed given
061         */
062        public static void turn( double speed )
063        {
064            RobotActuators.driveRight.set(-speed);
065            RobotActuators.driveLeft.set(speed);
066        }
067        
068        /**
069         * 
070         * stops all the motors
071         */
072        public static void stop()
073        {
074            RobotActuators.driveRight.set(0);
075            RobotActuators.driveLeft.set(0);
076        }
077        
078        /**
079         * switches gear when called
080         * @param ServoPosition used to set transmissionLeft position
081         */
082        public static void switchGear( double ServoPosition){
083    //        RobotActuators.transmissionLeft.set(ServoPosition);
084    //        System.out.println(RobotActuators.transmissionLeft.get());
085    
086            RobotActuators.transmissionLeft.set(ServoPosition);
087            RobotActuators.transmissionRight.set(ServoPosition);
088        }
089    }
